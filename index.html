Como Funciona

npm i @nexuspay/sdk

import { Nexus } from '@nexuspay/sdk'

const nx = new Nexus({ apiKey: 'SEU_API_KEY' })

const result = await nx.payments.charge({
  amount: 1990,
  currency: 'BRL',
  cardToken: 'tok_***',
  metadata: { orderId: 'NP-8421' }
})

console.log(result.status) // 'authorized'
